<div class="blog-filters">
  <button class="blog-filter active" data-filter="all">All</button>
  {% assign sorted_cats = site.categories | sort %}
  {% for category in sorted_cats %}
  <button class="blog-filter" data-filter="{{ category[0] | slugify }}">{{ category[0] }}</button>
  {% endfor %}
</div>

<div class="blog-posts">
  {% for post in site.posts %}
  <article class="blog-post-item" data-categories="{% for cat in post.categories %}{{ cat | slugify }} {% endfor %}">
    <div class="blog-post-meta">
      <time datetime="{{ post.date | date: "%Y-%m-%d" }}">{{ post.date | date: "%B %-d, %Y" }}</time>
      {% for cat in post.categories %}
      <span class="blog-post-cat">{{ cat }}</span>
      {% endfor %}
    </div>
    <h2><a href="{{ post.url | relative_url }}">{{ post.title | escape }}</a></h2>
    <p>{{ post.excerpt | strip_html | truncatewords: 30 }}</p>
    <a href="{{ post.url | relative_url }}" class="blog-read-more">Read More →</a>
  </article>
  {% endfor %}
</div>

<script>
(function () {
  var filters = document.querySelectorAll('.blog-filter');
  var posts   = document.querySelectorAll('.blog-post-item');

  filters.forEach(function (btn) {
    btn.addEventListener('click', function () {
      var filter = this.dataset.filter;
      filters.forEach(function (b) { b.classList.remove('active'); });
      this.classList.add('active');
      posts.forEach(function (post) {
        var cats = ' ' + post.dataset.categories + ' ';
        post.hidden = !(filter === 'all' || cats.indexOf(' ' + filter + ' ') !== -1);
      });
    });
  });
}());
</script>
